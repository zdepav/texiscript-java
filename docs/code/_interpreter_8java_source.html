122  private Generator buildGenerator(StGenerator expression) throws SemanticException {
123  if (expression instanceof StNumber) {
124  return Generator.get(((StNumber)expression).getValue());
125  } else if (expression instanceof StColor) {
126  return Generator.get(((StColor)expression).getValue());
127  } else if (expression instanceof StVariable) {
128  return scope.getConstant(expression.getCodePosition(), ((StVariable)expression).getName());
129  } else if (expression instanceof StFunctionCall) {
130  var fnCall = (StFunctionCall)expression;
131  var args = new Generator[fnCall.getArgumentCount()];
132  for (var i = 0; i < args.length; ++i) {
133  args[i] = buildGenerator(fnCall.getArgument(i));
134  }
135  var f = scope.getFunction(fnCall.getCodePosition(), fnCall.getName());
136  var g = f.call(fnCall.getCodePosition(), args);
137  if (g.isNumber()) {
138  return Generator.get(g.getDouble(0, 0));
139  } else if (g.isColor()) {
140  return Generator.get(g.getColor(0, 0));
141  } else {
142  g.setRandomSeed(fnCall.getSeed());
143  return g;
144  }
145  } else {
146  // should never happen - can only be caused by a bug in the parser
147  throw new RuntimeException();
148  }
