80  public Generator getConstant(CodePosition pos, String name) throws SemanticException {
81  Generator value;
82  int dotPos = name.lastIndexOf('.');
83  if (dotPos >= 0) {
84  // constants inside a package
85  var packageName = name.substring(0, dotPos);
86  name = name.substring(dotPos + 1);
87  if (!packages.containsKey(packageName)) {
88  throw new SemanticException(pos, "Package " + packageName + " does not exist.");
89  }
90  var pack = packages.get(packageName);
91  if ((value = pack.getValue(name)) != null) {
92  return value;
93  } else
94  throw new SemanticException(pos, "Constant " + name + " does not exist in package " + packageName + ".");
95  } else if ((value = packages.get("").getValue(name)) != null) {
96  // constant outside of any package (these take precedence before imported constants)
97  return value;
98  } else {
99  // imported constants
100  Generator g;
101  String packageName = null;
102  for (var pack: imported.entrySet()) {
103  if ((g = pack.getValue().getValue(name)) != null) {
104  if (packageName != null) {
105  throw new SemanticException(
106  pos,
107  "Name conflict between " + packageName + '.' + name + " and " +
108  pack.getKey() + '.' + name + ", use qualified name instead."
109  );
110  } else {
111  packageName = pack.getKey();
112  value = g;
113  }
114  }
115  }
116  if (value != null) {
117  return value;
118  }
119  }
120  // error - constant not found
121  for (var pack: packages.entrySet()) {
122  if (!pack.getKey().equals("") && pack.getValue().getValue(name) != null) {
123  throw new SemanticException(
124  pos,
125  "Constant " + name + " does not exist. Did you mean " +
